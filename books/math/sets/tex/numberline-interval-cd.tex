\fill[midblue] (-1,\yC) circle (2.2pt);

\node[color=midred, font=\normalsize, scale=\labelscale] at ($(-4,\yD)+(0,\labelgap)$) {D};
\node[color=midblue, font=\normalsize, scale=\labelscale] at ($(-2.5,\yC)+(0,\labelgap)$) {C};
