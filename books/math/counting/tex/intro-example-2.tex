\begin{document}
  \begin{tikzpicture}[x=1cm,y=1cm, line cap=round, line join=round]

    \tikzset{
      main/.style={draw=black, line width=0.85pt},
      vtx/.style={circle, draw=darkgray, fill=vtxfill, line width=0.85pt, minimum size=10pt, inner sep=0pt},
      vlab/.style={font=\Large},
    }

    \coordinate (A) at (0,0);
    \coordinate (B) at (4,0);
    \coordinate (C) at (8,0);

% parallel edges A--B
    \draw[main] (A) to[out=45,in=135]
    node[midway, above=1pt, darkred, font=\Large] {a}
    (B);

    \draw[main] (A) to[out=-45,in=-135]
    node[midway, above=1pt, darkred, font=\Large] {b}
    (B);

% parallel edges B--C (upper + straight + lower)
    \draw[main] (B) to[out=45,in=135]
    node[midway, above=1pt, darkred, font=\Large] {1}
    (C);

    \draw[main] (B) --
    node[midway, above=1pt, darkred, font=\Large] {2}
    (C);

    \draw[main] (B) to[out=-45,in=-135]
    node[midway, above=1pt, darkred, font=\Large] {3}
    (C);

% vertices
    \node[vtx] at (A) {};
    \node[vtx] at (B) {};
    \node[vtx] at (C) {};

% vertex labels
    \node[vlab, anchor=north] at ($(A)+(0,-0.35)$) {A};
    \node[vlab, anchor=north] at ($(B)+(0,-0.35)$) {B};
    \node[vlab, anchor=north] at ($(C)+(0,-0.35)$) {C};

  \end{tikzpicture}
